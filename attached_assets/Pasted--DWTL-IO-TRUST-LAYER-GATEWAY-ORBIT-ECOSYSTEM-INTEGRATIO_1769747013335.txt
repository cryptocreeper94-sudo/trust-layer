## STEP 2: CREATE ORBIT CLIENT FILE
Create file: `server/orbitClient.ts`
```typescript
// server/orbitClient.ts
// ORBIT Ecosystem Hub Client for Trust Layer Gateway (dwtl.io)
export class OrbitEcosystemClient {
  private apiKey: string;
  private apiSecret: string;
  private hubUrl = 'https://orbitstaffing.io';
  private appName = 'Trust Layer Gateway';
  constructor() {
    this.apiKey = process.env.ORBIT_HUB_API_KEY || '';
    this.apiSecret = process.env.ORBIT_HUB_API_SECRET || '';
  }
  private getHeaders(): Record<string, string> {
    return {
      'Content-Type': 'application/json',
      'X-API-Key': this.apiKey,
      'X-API-Secret': this.apiSecret,
      'X-App-Name': this.appName
    };
  }
  // Test connection to ORBIT Hub
  async checkConnection(): Promise<{ connected: boolean; hubName?: string; error?: string }> {
    try {
      const res = await fetch(`${this.hubUrl}/api/ecosystem/status`, {
        method: 'GET',
        headers: this.getHeaders()
      });
      
      if (!res.ok) {
        const error = await res.json();
        return { connected: false, error: error.error || 'Connection failed' };
      }
      
      const data = await res.json();
      return { connected: true, hubName: data.hubName };
    } catch (error) {
      return { connected: false, error: String(error) };
    }
  }
  // Log activity to ecosystem hub
  async logActivity(action: string, category: string, details: Record<string, any>): Promise<boolean> {
    try {
      const res = await fetch(`${this.hubUrl}/api/ecosystem/logs`, {
        method: 'POST',
        headers: this.getHeaders(),
        body: JSON.stringify({
          action,
          category,
          details,
          timestamp: new Date().toISOString()
        })
      });
      return res.ok;
    } catch {
      return false;
    }
  }
  // Share code snippet with ecosystem
  async shareSnippet(title: string, code: string, language: string, description?: string): Promise<any> {
    try {
      const res = await fetch(`${this.hubUrl}/api/ecosystem/snippets`, {
        method: 'POST',
        headers: this.getHeaders(),
        body: JSON.stringify({ title, code, language, description })
      });
      return await res.json();
    } catch (error) {
      return { success: false, error: String(error) };
    }
  }
  // Get shared snippets from ecosystem
  async getSnippets(): Promise<any[]> {
    try {
      const res = await fetch(`${this.hubUrl}/api/ecosystem/snippets`, {
        headers: this.getHeaders()
      });
      const data = await res.json();
      return data.snippets || [];
    } catch {
      return [];
    }
  }
}
// Export singleton instance
export const orbitHub = new OrbitEcosystemClient();
